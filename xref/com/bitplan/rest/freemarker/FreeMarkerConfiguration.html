10  package com.bitplan.rest.freemarker;
11  
12  import java.io.File;
13  import java.io.IOException;
14  import java.io.StringWriter;
15  import java.util.ArrayList;
16  import java.util.List;
17  import java.util.Map;
18  
19  import freemarker.cache.ClassTemplateLoader;
20  import freemarker.cache.FileTemplateLoader;
21  import freemarker.cache.MultiTemplateLoader;
22  import freemarker.cache.TemplateLoader;
23  import freemarker.ext.beans.BeansWrapper;
24  import freemarker.template.Template;
25  
26  /**
27   * supply a free marker configuration
28   * @author wf
29   *
30   */
31  public class FreeMarkerConfiguration {
32  	
33  	protected static freemarker.template.Configuration cfg = null;
34  
35  	// multiple template pathes for files possible
36  	protected static List<String> templatePaths=new ArrayList<String>();
37  	
38  	// list of classes to load templates from
39  	protected static List<TemplateClass> templateClasses=new ArrayList<TemplateClass>();
40  	
41  	/**
42  	 * add a path to the template path
43  	 * @param path - the path to add 
44  	 */
45  	public static void addTemplatePath(String path) {
46  		templatePaths.add(path);
47  		// force configuration refresh
48  		cfg=null;
49  	}
50  	
51  	/**
52  	 * add the given template class from the given clazz
53  	 * @param clazz the Class to add
54  	 * @param path the path to use
55  	 */
56  	@SuppressWarnings("rawtypes")
57  	public static void addTemplateClass(Class clazz, String path) {
58  		TemplateClass tClass=new TemplateClass();
59  		tClass.clazz=clazz;
60  		tClass.path=path;
61  		templateClasses.add(tClass);
62  		// force configuration refresh
63  		cfg=null;
64  	}
65  	
66  	/**
67  	 * get the freemarker configuration
68  	 * @return the configuration
69  	 * @throws IOException if an I/O problem occurs
70  	 */
71  	public static freemarker.template.Configuration getFreemarkerConfiguration() throws IOException {
72  		if (cfg == null) {
73  			// http://freemarker.sourceforge.net/docs/pgui_quickstart_createconfiguration.html
74  			cfg = new freemarker.template.Configuration();
75  			// Specify the data source where the template files come from.
76  			// use multiple sources
77  			List<TemplateLoader> loaderList=new ArrayList<TemplateLoader>();
78  			for (String templatePath:templatePaths) {
79  				loaderList.add(new FileTemplateLoader(new File(templatePath)));
80  			}
81  			for (TemplateClass templateClass:templateClasses) {
82  				loaderList.add(new ClassTemplateLoader(templateClass.clazz,templateClass.path));
83  			}
84  			TemplateLoader[] loaders = new TemplateLoader[loaderList.size()];
85  			loaderList.toArray(loaders);
86  			MultiTemplateLoader mtl = new MultiTemplateLoader(loaders);
87  
88  			cfg.setTemplateLoader(mtl);  
89  			
90  			// Specify how templates will see the data-model. This is an advanced
91  			// http://freemarker.sourceforge.net/docs/api/freemarker/ext/beans/BeansWrapper.html
92  			BeansWrapper wrapper = new BeansWrapper();
93  			wrapper.setSimpleMapWrapper(true);
94  			cfg.setObjectWrapper(wrapper);
95  			// http://freemarker.org/docs/app_faq.html#faq_number_grouping
96  			cfg.setNumberFormat("0.######"); 
97  		}
98  		return cfg;
99  	}
100 
101 	/**
102 	 * get the Template with the given name
103 	 * @param templateName - the name of the template to look for
104 	 * @return - the matching Template
105 	 * @throws IOException  - in case of an I/O problem
106 	 */
107 	public static Template getTemplate(String templateName) throws IOException {
108 		Template result = getFreemarkerConfiguration().getTemplate(templateName);
109 		return result;
110 	}
111 
112 	/**
113 	 * process the given template with the given map
114 	 * @param templateName - the name of the template to use
115 	 * @param rootMap - the map to pass on to Freemarker's template engine
116 	 * @return a String with the result of processing the template
117 	 * @throws Exception - in case of failure
118 	 */
119 	public static String doProcessTemplate(String templateName,Map<String,Object> rootMap) throws Exception {
120 		Template template=FreeMarkerConfiguration.getTemplate(templateName);
121 		StringWriter htmlWriter = new StringWriter();
122 		template.process(rootMap,htmlWriter);
123 		return htmlWriter.toString();
124 	}
125 }
