32  import freemarker.template.Template;

35   * test jqGridFilter
36   * @author wf

38   */
39  public class TestFreemarker {
40  
41  	@Test
42  	public void testFreemarker() throws Exception {
43  		// add a template path (test.ftl is to be loaded from this path)
44  		FreeMarkerConfiguration.addTemplatePath("test/template");
45  		// add a class path (test1.ftl is to be loaded from this path the source is
46  		// at src/main/resources/test1.ftl and it is included via
47  		// target/main/resources/test1.ftl 
48  		FreeMarkerConfiguration.addTemplateClass(FreeMarkerConfiguration.class, "/templates");
49  		// the list of test template names
50  		String templateNames[]={"test.ftl","test1.ftl"};
51  		Map<String,Object> rootMap=new HashMap<String,Object>();
52  		String testTitle="Freemarker rocks!";
53  		rootMap.put("title", testTitle);
54  		// loop over the templates
55  		for (String templateName:templateNames){
56  			// get the template by templateName no matter what the path is
57  			Template template = FreeMarkerConfiguration.getTemplate(templateName);
58  			// make sure the template is found
59  			assertNotNull("template "+templateName+" should not be null",template);
60  			String html=FreeMarkerConfiguration.doProcessTemplate(templateName, rootMap);
61  			assertTrue(html.contains(testTitle));
62  		}
63  	}
