175 	/**
176 	 * check the given html text
177 	 * @param html
178 	 * @param expectedTagCount
179 	 */
180 	public void checkHtml(String html, int expectedTagCount) {
181 		CleanerProperties props = new CleanerProperties();
182 		// set some properties to non-default values
183 		props.setTranslateSpecialEntities(true);
184 		props.setTransResCharsToNCR(true);
185 		props.setOmitComments(true);
186 		 
187 		// do parsing
188 		TagNode tagNode = new HtmlCleaner(props).clean(
189 		    html
190 		);
191 		TagNode[] tags = tagNode.getAllElements(true);
192 		if (debug) {
193 			System.out.println("found: "+tags.length+" tags");
194 		}
195 		assertEquals("html #of tags",expectedTagCount,tags.length);
196 	}
